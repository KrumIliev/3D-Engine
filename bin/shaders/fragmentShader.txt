#version 400 core

in vec2 pass_textureCoords;
in vec3 serfaceNormal;
in vec3 toLightVector;

out vec4 out_Color;

uniform sampler2D textureSampler;
uniform vec3 lightColor;

void main (void) {

	vec3 utilNormal = normalize(serfaceNormal);
	vec3 utilLightVector = normalize(toLightVector);
	
	float nDot1 = dot(utilNormal, utilLightVector);
	float brightness = max(nDot1, 0.0);
	vec3 diffuse = brightness * lightColor;

	out_Color = vec4(diffuse, 1.0) * texture(textureSampler, pass_textureCoords);
	
}
